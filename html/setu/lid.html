196    def char_percent_check(self, input):
197        # check whether input has input_threhsold of roman characters
198        
199        # count total number of characters in string
200        input_len = len(list(input))
201
202        # count special character spaces and newline in string
203        special_char_pattern = re.compile('[@_!#$%^&*()<>?/\|}{~:]')
204        special_char_matches = special_char_pattern.findall(input)
205        special_chars = len(special_char_matches)
206        
207        spaces = len(re.findall('\s', input))
208        newlines = len(re.findall('\n', input))
209
210        # subtract total-special character counts
211        total_chars = input_len - (special_chars + spaces + newlines)
212
213        # count the number of english character and digit in string
214        en_pattern = re.compile('[a-zA-Z0-9]')
215        en_matches = en_pattern.findall(input)
216        en_chars = len(en_matches)
217
218        # calculate the percentage of english character in total number of characters
219        if total_chars == 0:
220            return 0
221        return (en_chars/total_chars)

324    def batch_predict(self, input_list, batch_size):
325
326        # call functions seq by seq and divert the input to IndicBERT if 
327        # fasttext prediction score is less than the defined model_threhsold.
328        # Also output the inference time along with the result.
329        output_dict = {}

334        # text roman percent check 
335        for index, input in enumerate(input_list):
336            if self.char_percent_check(input) > self.input_threshold:

343        
344        results = self.post_process(output_dict)

470    def hard_vote(self, indiclid_res, cld3_res, nllb_res, lid_probability_threshold=None):
471
472        if lid_probability_threshold:
473           self.lid_probability_threshold = lid_probability_threshold
474        
475        overall_res = indiclid_res + cld3_res + nllb_res
476        language_vote = {}
477        detector_mapping = {0: "indiclid", 1: "cld3", 2: "nllb"}
478
479        for i in range(len(overall_res)):
480            res = overall_res[i]
481            if res["language"] not in language_vote.keys():
482                language_vote[res["language"]] = {}
483
484            language_vote[res["language"]][detector_mapping[i]] = res["probability" if "probability" in tuple(res.keys()) else "logits"]
485
486            # if probability is there (like cld3 & NLLB), then it should be above the threshold to be considered, 
487            # else if probability is not present at all (like IndicLID), then we will always consider it for voting.
488            if "probability" in tuple(res.keys()) and res["probability"] >= self.lid_probability_threshold: 
489                language_vote[res["language"]]["total_votes"] = language_vote[res["language"]].get("total_votes", 0) + 1
490            else:
491                language_vote[res["language"]]["total_votes"] = language_vote[res["language"]].get("total_votes", 0) + 1
492        
493        lang_vote_pair = list(language_vote.items())
494        lang_vote_pair.sort(reverse=True, key=lambda x: x[1]["total_votes"])
495        majority_lang = lang_vote_pair[0]
496        return majority_lang, language_vote

197    def char_percent_check(self, input):
198        # check whether input has input_threhsold of roman characters
199        
200        # count total number of characters in string
201        input_len = len(list(input))
202
203        # count special character spaces and newline in string
204        special_char_pattern = re.compile('[@_!#$%^&*()<>?/\|}{~:]')
205        special_char_matches = special_char_pattern.findall(input)
206        special_chars = len(special_char_matches)
207        
208        spaces = len(re.findall('\s', input))
209        newlines = len(re.findall('\n', input))
210
211        # subtract total-special character counts
212        total_chars = input_len - (special_chars + spaces + newlines)
213
214        # count the number of english character and digit in string
215        en_pattern = re.compile('[a-zA-Z0-9]')
216        en_matches = en_pattern.findall(input)
217        en_chars = len(en_matches)
218
219        # calculate the percentage of english character in total number of characters
220        if total_chars == 0:
221            return 0
222        return (en_chars/total_chars)
